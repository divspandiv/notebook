$('.phone-number').on('input keyup change',function (e) {

	let txt = $(this).val();
	    txt = txt.replace(/\s*/g, "");
	    let result = [];
	    for(let i = 0; i < txt.length; i++){
	        if ((i>3?((i-3)%4):i%3) == 0 && i != 0){
	          result.push(" " + txt.charAt(i));
	        } else {
	          result.push(txt.charAt(i));
	        }
	    }
	    txt = result.join("");
	    if(txt.length>=13){
		txt = txt.substring(0,13);
	    }
            $(this).val(txt);
}
